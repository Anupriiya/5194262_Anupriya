  int isLeap(int y) {
  return (y % 400 == 0) || ((y % 4 == 0) && (y % 100 != 0));
    }
    void nextDate(int *d, int *m, int *y) {
    int daysInMonth[] = {0,31,28,31,30,31,30,31,31,30,31,30,31};
    if (isLeap(*y)) daysInMonth[2] = 29;

    (*d)++;
    if (*d > daysInMonth[*m]) {
        *d = 1;
        (*m)++;
        if (*m > 12) {
            *m = 1;
            (*y)++;
        }
    }
}
int findPrimeDates(int d1, int m1, int y1, int d2, int m2, int y2) {
    int count = 0;

    while (1) {
        long long z = (long long)d1*1000000 + (long long)m1*10000 + y1;

        if (z % 4 == 0 || z % 7 == 0)
            count++;

        if (d1 == d2 && m1 == m2 && y1 == y2) break;
        nextDate(&d1, &m1, &y1);
    }

    return count;
    
}


    

    
int main() {
    char start[11], end[11];
    scanf("%s %s", start, end);

    int d1, m1, y1, d2, m2, y2;
    sscanf(start, "%d-%d-%d", &d1, &m1, &y1);
    sscanf(end, "%d-%d-%d", &d2, &m2, &y2);

    printf("%d", findPrimeDates(d1, m1, y1, d2, m2, y2));
    return 0;

    
}
